Задача нахождения собственных значений и собственных векторов для матрицы заключается в поиске таких соответствующих друг другу чисел и ненулевых векторов которые удовлетворяют уравнению $Ax = \lambda x$, при этом числа $\lambda$ называются собственными значениями, а вектора $x$ собственными векторами.

Наиболее применяемым алгоритмом для вычисления собственных чисел матрицы является QR-алгоритм, который приводит матрицу к блочному верхнетреугольному виду. В результате на диагонали лежат блоки 1 на 1 (вещественные числа) и блоки 2 на 2 (комплексные числа).
![[{99E3633F-5414-44CA-84DF-5292399F89EB}.png]]

## Математическое описание алгоритма
Известно, что произвольная квадратная матрица может быть представлена в виде произведения унитарной (в вещественном случае ортогональной) и верхней треугольной матриц. Такое разложение называется QR-разложением.
Пусть A - исходная матрица. Для различных k выполняется разложение
![[{ECB596B1-2B2B-47B2-9A59-A35FBA7E0D10}.png]]
Все матрицы Ak унитарно подобны, то есть их собственные значения равны.

Для ускорения сходимости QR-алгоритма используются два основных подхода:
+ приведение изначальной матрицы к форме Хессенберга
+ использование сдвигов во время итераций

Приведение матрицы к форме Хессенберга уменьшает количество зануляемых элементов, так как хессенбергова форма - почти треугольная, QR-алгоритму остается занулить только одну поддиагональ под главной диагональю.

Сдвиги сокращают количество итераций QR-алгоритма. Например, в сдвиге Вилкинсона используется нижний левый блок (2 на 2) для определения величины сдвига, на которую уменьшаются все значения главной диагонали. Затем матрица восстанавливается путем добавления этого сдвига после итерации QR-алгоритма.


## Алгоритм
```
    if(m.columns != m.rows){
        std::cout << "srectrum(): matrix must be square" << std::endl;
        return m;
    }

    Matrix<double> R = hessenberg(m); //верхнетреугольная матрица
    Matrix<double> Q(R.rows, R.columns); //ортогональная матрица
    Matrix<double> G(R.rows, R.columns); //Матрица вращения

    double r, c, s, mu;
    unsigned int i;
    bool converged = true;
    while(1){
        Q = Q.identity();

        //Сдвиг Вилкинсона
        mu = wilkinson_shift(R);
        for (i=0; i<R.rows; i++) R.data[i][i] -= mu;

        for(i=0; i<R.columns-1; i++){
            //Если элемент уже ноль - пропускаем
            if(fabs(R.data[i+1][i]) < EPSILON) continue;

            //Норма вектора 2 на 1 под главной диагональю
            r = sqrt(R.data[i][i]*R.data[i][i] + R.data[i+1][i]*R.data[i+1][i]);

            //Коэффициенты вращения
            c = R.data[i][i] / r;
            s = R.data[i+1][i] / r;

            //Матрица вращения
            G = G.identity();
            G.data[i][i] = c;
            G.data[i][i+1] = -s;
            G.data[i+1][i] = s;
            G.data[i+1][i+1] = c;

            //Применяем вращение
            R = G.transpose() * R;

            //Ортогональная матрица
            Q = Q * G;
        }

  

        //Матрица для следующей итерации
        R = R * Q;
        for (i=0; i<R.rows; i++) R.data[i][i] += mu;
        R.clean();

        //Проверка сходимости
        converged = true;

        for (i=0; i<R.columns-1; i++) {
            if (fabs(R.data[i+1][i]) < EPSILON) continue;

            //Блок 2 на 2, представляющий комплексное число [a, b][c, d]
            double a = R.data[i][i];
            double b = R.data[i][i+1];
            double c = R.data[i+1][i];
            double d = R.data[i+1][i+1];
            double disc = (a - d)*(a - d) + 4*b*c;

            if (disc < 0.0) {// комплексная пара
                i++;
                continue;
            }

            converged = false;
            break;
        }

        if (converged) break;
    }

    return R;
```

После работы алгоритма получаем матрицу с блоками 1 на 1 (вещественные числа) и с блоками 2 на 2 (комплексные числа) на главной диагонали. Блоки 2 на 2 интерпретируются в комплексные числа следующим образом:
```
    Matrix<double> out{QR(m)};

    unsigned int i, ii;
    //зануляем ненужные элементы
    for(i=0; i<out.rows; i++){
        for(ii=0; ii<out.columns; ii++){
            if(abs((int)i-(int)ii)<=1) continue; //не изменяем главную диагональ и две соседние поддиагонали
            out.data[i][ii] = 0;
        }
    }

    //зануляем ненужные элементы рядом с главной диагональю
    for(i=0; i<out.columns-1; i++){
        if(out.data[i+1][i]==0) out.data[i][i+1] = 0;
    }

    double a, b, c, d, trace, det, discriminant;
    //вычисляем комплексные собственные числа из блоков 2 на 2 на главной диагонали
    for(i=0; i<out.columns-1; i++){
        if(out.data[i+1][i]==0) continue; //проверка значений под главной диагональю

        a = out.data[i][i];
        b = out.data[i][i+1];
        c = out.data[i+1][i];
        d = out.data[i+1][i+1];

        trace = a + d;
        det = a*d - b*c;
        discriminant = trace*trace - 4*det;

        if(discriminant>=0){ //два вещественных числа
            out.data[i][i] = (trace + sqrtf(discriminant)) / 2;
            out.data[i][i+1] = 0;
            out.data[i+1][i] = 0;
            out.data[i+1][i+1] = (trace - sqrtf(discriminant)) / 2;
        }else{ //два комплексно-сопряженных числа
            out.data[i][i] = trace / 2; //действительная часть
            out.data[i][i+1] = sqrtf(-discriminant) / 2; // + мнимая часть
            out.data[i+1][i] = -out.data[i][i+1]; // - мнимая часть
            out.data[i+1][i+1] = 0;
        }

        i++; //пропускаем элемент [i+1][i+1]
        continue;
    }

    return out;
```
