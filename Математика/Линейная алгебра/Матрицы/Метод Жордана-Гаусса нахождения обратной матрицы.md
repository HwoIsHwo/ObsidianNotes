Метод заключается в построении расширенной матрицы путем приписывания справа к изначальной матрице единичной. Затем левая часть (изначальная матрица) приводится к виду единичной, тогда на месте изначальной единичной матрицы будет обратная.

1. Изначальная матрица:

$$
\begin{pmatrix}  
1.07 & -3 \\
2.2 & 0.1
\end{pmatrix}
$$

2. Приписываем единичную
$$
\begin{pmatrix}
1.07  &  -3  & | &  1   &    0 \\
2.2   &  0.1  & | &  0   &    1
\end{pmatrix}
$$

3. Преобразуем левую часть в единичную матрицу
$$
\begin{pmatrix}
1    &   -2.80374  & | &      0.934579    &    0 \\
0     &  6.26822 &|&-2.05607     &   1
\end{pmatrix}
$$
$$
\begin{pmatrix}
1    &   0  & | &     0.0149098   &    0.447294 \\
0    &   1   &  | & -0.328016    &   0.159535
\end{pmatrix}
$$
4. Обратная матрица:
$$
\begin{pmatrix}
0.0149098   &    0.447294 \\
-0.328016    &   0.159535
\end{pmatrix}
$$

Пример реализации на языке C++:
```
template <typename T>
Matrix<T> Matrix<T>::operator!() `const`
{
    if(rows!=columns){
        std::cout << "operator !: the matrix is ​​not square" << std::endl;
        return *this;
    }
    
    Matrix<T> expand = merge(identity());
    
    unsigned int i, ii, iii;
    T a0, a1;
    //В каждом столбце исходной матрицы
    for(i=0; i<columns; i++){
        //Делаем диагональный элемент в этом столбце = 1
        a0 = expand.data[i][i];
        for(ii=0; ii<expand.columns; ii++){
            expand.data[i][ii] = expand.data[i][ii] / a0;
        }

        //Для каждого элемента в столбце
        for(ii=0; ii<rows; ii++){
            //если не диагональный элемент этого столбца
            if(i!=ii){
                //зануляем при помощи строки с диагональным элементом из этого столбца
                a1 = expand.data[ii][i];
                for(iii=0; iii<expand.columns; iii++){
                    expand.data[ii][iii] = expand.data[ii][iii] - expand.data[i][iii] * a1;
                }
            }
        }
    }
    
    return expand;
}
```
